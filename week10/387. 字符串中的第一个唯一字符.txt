class Solution {
public:
    int firstUniqChar(string s) {
        unordered_map<char, int> chToCount;
        for (const auto& c : s) {
            chToCount[c]++;
        }

        for (int i = 0; i < s.size(); i++) {
            if (chToCount[s[i]] == 1) {
                return i;
            }
        }

        return -1;
    }
};
